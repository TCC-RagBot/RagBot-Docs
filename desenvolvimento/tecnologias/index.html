# Modelo otimizado para portugu√™s e ingl√™s

def generate_embedding(text: str) -> List[float]:
    """
    Gera embedding de 384 dimens√µes para o texto.

    Args:
        text: Texto para gerar embedding

    Returns:
        Lista de floats representando o vetor

# Busca por similaridade
def semantic_search(query: str, chunks: List[str], top_k: int = 5):
    query_embedding = model.encode([query])
    chunk_embeddings = model.encode(chunks)

    similarities = cosine_similarity(query_embedding, chunk_embeddings)[0]
    top_indices = similarities.argsort()[-top_k:][::-1]

    return [(chunks[i], similarities[i]) for i in top_indices]

    Extrai texto de um arquivo PDF.

    Args:
        pdf_bytes: Conte√∫do bin√°rio do PDF
